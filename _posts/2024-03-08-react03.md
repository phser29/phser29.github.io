const handleCart = () => {
  const cartItem = {
    id: product.id,
    image: product.image,
    name: product.name,
    price: product.price,
    provider: product.provider,
    quantity: count
  }
  // 장바구니 cartItem와 일치하는 cart.id가 있으면 
  const found = cart.find((el) => el.id === cartItem.id);
  if(found) {
    setQuantity(cartItem.id, found.quantity + count);
  } else {
    setCart([...cart, cartItem]);
  }
}
